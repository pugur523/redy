if(ENABLE_BUILD_INFO_DEFINITIONS)
  set(BUILD_NAME ${MAIN_EXECUTABLE_NAME_FROM_CONFIG} CACHE STRING "" FORCE)
  set(BUILD_VERSION ${PROJECT_VERSION} CACHE STRING "" FORCE)
  set(BUILD_NUMBER "0" CACHE STRING "" FORCE)
  set(BUILD_DESCRIPTION ${PROJECT_DESCRIPTION} CACHE STRING "" FORCE)
  set(BUILD_TYPE ${CMAKE_BUILD_TYPE} CACHE STRING "" FORCE)
  string(TIMESTAMP BUILD_TIME "%Y-%m-%d %H:%M UTC" UTC)
  set(BUILD_PLATFORM ${CMAKE_HOST_SYSTEM_NAME} CACHE STRING "" FORCE)
  set(BUILD_ARCH ${CMAKE_HOST_SYSTEM_PROCESSOR} CACHE STRING "" FORCE)
  set(BUILD_COMPILER "${CMAKE_CXX_COMPILER_ID} (version ${CMAKE_CXX_COMPILER_VERSION})" CACHE STRING "" FORCE)

  set(TARGET_PLATFORM ${CMAKE_SYSTEM_NAME} CACHE STRING "" FORCE)
  set(TARGET_ARCH ${CMAKE_SYSTEM_PROCESSOR} CACHE STRING "" FORCE)

  # TARGET_BITS
  if(CMAKE_SIZEOF_VOID_P EQUAL 8)
    set(TARGET_BITS "64 bit" CACHE STRING "" FORCE)
  elseif(CMAKE_SIZEOF_VOID_P EQUAL 4)
    set(TARGET_BITS "32 bit" CACHE STRING "" FORCE)
  else()
    set(TARGET_BITS "unknown" CACHE STRING "" FORCE)
  endif()

  # BUILD_GIT_COMMIT_HASH
  find_package(Git QUIET)

  if(GIT_FOUND)
    execute_process(
      COMMAND ${GIT_EXECUTABLE} rev-parse --short HEAD
      WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
      OUTPUT_VARIABLE BUILD_GIT_COMMIT_HASH
      OUTPUT_STRIP_TRAILING_WHITESPACE
      RESULT_VARIABLE GIT_RESULT
    )

    if(NOT GIT_RESULT EQUAL 0)
      set(BUILD_GIT_COMMIT_HASH "unknown" CACHE STRING "" FORCE)
    endif()
  else()
    set(BUILD_GIT_COMMIT_HASH "unknown" CACHE STRING "" FORCE)
  endif()
endif()

configure_file(
  "${CMAKE_CURRENT_SOURCE_DIR}/build_config.h.in"
  "${CMAKE_CURRENT_SOURCE_DIR}/build_config.h"
  @ONLY
)
